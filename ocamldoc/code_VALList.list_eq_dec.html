let rec list_eq_dec eq_dec l l' =
  match l with
  | Coq_nil ->
    (match l' with
     | Coq_nil -> Coq_left
     | Coq_cons (y, l'0) -> Coq_right)
  | Coq_cons (y, l0) ->
    (match l' with
     | Coq_nil -> Coq_right
     | Coq_cons (y0, l'0) ->
       let s = eq_dec y y0 in
       (match s with
        | Coq_left -> list_eq_dec eq_dec l0 l'0
        | Coq_right -> Coq_right))
